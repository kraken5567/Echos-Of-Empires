country_event = {
	id = USA_industry.3
	title = "A New Building!"
	desc = "We have been given more industry through the power of Laissez-Faire!"
	trigger = { }
	is_triggered_only = yes

	option = {
		name = "Free Market is a happy market!"
		if = {
			limit = {
				check_variable = {
					var = random
					value = 0.5
					compare = less_than_or_equals
				}
			}
			random_core_state = {
				add_extra_state_shared_building_slots = 1
				add_building_construction = {
					type = industrial_complex
					level = 1
				}
			}
		}
		else = {
			random_core_state = {
				add_extra_state_shared_building_slots = 1
				add_building_construction = {
					type = arms_factory
					level = 1
				}
			}
		} 
		# about 4 months
		country_event = { id = USA_industry.3 days = 122 }
	}
}
