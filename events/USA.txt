#DESERET

country_event = {
	id = USA_interact_Ai.010
	title = "United States Proposes Contenetal Unity"
	desc = "How shall we handle this?"

	trigger = {
	}

	is_triggered_only = yes

	mean_time_to_happen = {
		days = 1 # 
	}

	option = {
		name = "Ai does a thing"
		if = {
			limit = {
				ROOT = { is_ai = yes }
				ROOT = {
					strength_ratio = {
						tag = USA
						ratio < 1.25
					}
				}
			}
			USA = {
				country_event = {id = USA.011 days = 7}
				}
		}
		else ={
			USA = {
				country_event = {id = USA.010 days = 7}
				}
		}
	}
} 

country_event = {
	id = USA_interact_Player.010
	title = "United States Proposes Contenetal Unity"
	desc = "How shall we handle this?"

	trigger = {
	}

	is_triggered_only = yes

	mean_time_to_happen = {
		days = 1 # 
	}

	option = {
		name = "Accept."
		USA = {
			country_event = {id = USA.011 days = 7}
			}
		}

	option = {
		name = "Reject."
		USA = {
			country_event = {id = USA.010 days = 7}
		}
	}
} 

country_event = {
	id = USA.010
	title = "Deseret Rejects our Proposal"
	desc = ""
	trigger = {
		
	}
	is_triggered_only = yes
	option = {
		name = "We tried"
		add_opinion_modifier = {
			target = Z83
			modifier = medium_decrease
		}
	}
	option = {
		name = "We will run over them!"
		add_opinion_modifier = {
			target = MEX
			modifier = large_decrease
		}
		Z83 = {
			every_core_state = {
				add_claim_by = USA
			}
		}
		create_wargoal = {
			target = Z83
			type = annex_everything
		}
	}
}

country_event = {
	id = USA.011
	title = "Deseret Accepts our Proposal"
	desc = ""
	trigger = {
		
	}
	is_triggered_only = yes
	option = {
		name = "Good"
		
		Z83 = {
			every_core_state = {
				add_core_of = USA
			}
		}
		USA = { 
			annex_country = { 
				target = Z83 
			} 
		}
	}
}
